from flask import Flask, request, render_template
from EmotionDetection.emotion_detection import emotion_detector

app = Flask(__name__)

@app.route("/emotionDetector")
def detect_emotion():
    """
    Analyzes the input text for emotions and returns a formatted string.
    Handles invalid or blank input by returning an error message.
    """
    # Retrieve the text from the query parameter
    text_to_analyze = request.args.get("textToAnalyze", "")
    
    # Call the emotion_detector function
    result = emotion_detector(text_to_analyze)

    # Task 7: Incorporate Error handling for blank entries/None dominant_emotion
    if result["dominant_emotion"] is None:
        return "Invalid text! Please try again!"

    # Task 3 & 4: Format the output for the interface
    return (
        f"For the given statement, the system response is "
        f"'anger': {result['anger']}, 'disgust': {result['disgust']}, "
        f"'fear': {result['fear']}, 'joy': {result['joy']} and "
        f"'sadness': {result['sadness']}. "
        f"The dominant emotion is {result['dominant_emotion']}."
    )

@app.route("/")
def render_index_page():
    """
    Renders the main application page.
    """
    return render_template('index.html')

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
